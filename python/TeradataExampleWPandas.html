Make a connection to Teradata and get some data. (I multiplied the actual "Bad_Passwd" by 100 to get some values to show up.) Then do a simple line graph with Pandas.

import teradata
import pandas as pd

udaExec = teradata.UdaExec()

qry1='''sel logdate LogDate
,sum(case when event='Logon' then 1 else 0 end ) Good_Logons
,sum(case when event='Bad Password' then 100 else 0 end) Bad_Passwd
from dbc.logonoff
group by 1
order by 1
;'''

       
conn = udaExec.connect("${dataSourceDev}")        
#df = pd.read_sql(qry1, conn,  columns = ['logdate', 'Success','Failure'])
df = pd.read_sql(qry1, conn,  index_col = ['LogDate'], columns = [ 'Success','Failure'])


df.plot();
